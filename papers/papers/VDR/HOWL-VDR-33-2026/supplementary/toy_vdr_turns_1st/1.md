```python
"""
Toy LLM data preparation.

    from data import build_dataset

    windows, vocab, inv_vocab, vocab_size = build_dataset("the cat sat on the mat", 4)
"""

from vdr.core import VDR
from vdr.linalg import Vec


def build_vocab(corpus):
    """
    Build vocabulary from corpus text.

    I: text string (space-separated tokens)
    O: dict {token: id} ordered by first appearance

        build_vocab("the cat sat on the mat")
        -> {"the": 0, "cat": 1, "sat": 2, "on": 3, "mat": 4}
    """
    vocab = {}
    for word in corpus.split():
        if word not in vocab:
            vocab[word] = len(vocab)
    return vocab


def invert_vocab(vocab):
    """
    Reverse vocabulary: {id: token}.

    I: vocab dict
    O: inverse dict
    """
    return {v: k for k, v in vocab.items()}


def tokenize(text, vocab):
    """
    Convert text to token ids.

    I: text string, vocab dict
    O: list of ints

        tokenize("the cat sat", vocab) -> [0, 1, 2]
    """
    return [vocab[w] for w in text.split()]


def detokenize(ids, inv_vocab):
    """
    Convert token ids back to text.

    I: list of ints, inverse vocab
    O: space-joined string
    """
    return " ".join(inv_vocab[i] for i in ids)


def make_windows(ids, seq_len):
    """
    Create sliding window training examples.

    Each window: (context of seq_len tokens, next token as target).

    I: full token id sequence, context length
    O: list of (context_ids, target_id)

        make_windows([0,1,2,3,0,4], 4)
        -> [([0,1,2,3], 0), ([1,2,3,0], 4)]
    """
    windows = []
    for i in range(len(ids) - seq_len):
        context = ids[i:i + seq_len]
        target = ids[i + seq_len]
        windows.append((context, target))
    return windows


def one_hot_target(target_id, vocab_size):
    """
    Exact one-hot vector.

    I: target index, vocabulary size
    O: Vec with VDR(1) at target, VDR(0) elsewhere

        one_hot_target(2, 5) -> Vec([0, 0, 1, 0, 0])
    """
    data = [VDR(0)] * vocab_size
    data[target_id] = VDR(1)
    return Vec(data)


def build_dataset(corpus, seq_len):
    """
    Build complete dataset from corpus text.

    I: corpus string, sequence length
    O: (windows, vocab, inv_vocab, vocab_size)

        windows, vocab, inv_vocab, vs = build_dataset("the cat sat on the mat", 4)
    """
    vocab = build_vocab(corpus)
    inv_vocab = invert_vocab(vocab)
    ids = tokenize(corpus, vocab)
    windows = make_windows(ids, seq_len)
    return windows, vocab, inv_vocab, len(vocab)
```

## File 3: `examples/toy_llm/model.py`

```python
"""
Toy LLM model — 1-layer decoder-only transformer.

    from model import ToyTransformer
    from config import *

    model = ToyTransformer()
    logits = model.forward([0, 1, 2, 3])
    # logits is list of Vec, one per position

All components use exact VDR arithmetic.
Every attention weight row sums to exactly 1.
Every intermediate is an exact rational.
"""

from vdr.core import VDR
from vdr.linalg import Vec, Mat
from vdr.ml.nn import Linear, ReLU, VecParam, MatParam
from vdr.ml.softmax import softmax
from vdr.ml.init import xavier_like_mat, zero_bias, rational_uniform_mat
from vdr.ml.attention import (
    attention_scores, causal_mask, apply_boolean_mask,
    attention_weights as compute_attention_weights,
    attention_mix, self_attention,
)

import config as cfg


class ToyTransformer:
    """
    1-layer decoder-only transformer with exact VDR arithmetic.

    226 parameters. Every one an exact rational.

        model = ToyTransformer()
        logits = model.forward([0, 1, 2, 3])
    """

    def __init__(self, seed=None):
        if seed is None:
            seed = cfg.SEED
        dim = cfg.DIM
        ffn_dim = cfg.FFN_DIM
        vocab_size = cfg.VOCAB_SIZE
        seq_len = cfg.SEQ_LEN
        denom = cfg.INIT_DENOM

        # Token embedding: vocab_size x dim
        emb_mat = rational_uniform_mat(
            vocab_size, dim, denom=denom, seed=seed, lo=-1, hi=1
        )
        self.token_emb = [
            Vec([emb_mat[i, j] for j in range(dim)])
            for i in range(vocab_size)
        ]

        # Positional embedding: seq_len x dim
        pos_mat = rational_uniform_mat(
            seq_len, dim, denom=denom, seed=seed + 1, lo=-1, hi=1
        )
        self.pos_emb = [
            Vec([pos_mat[i, j] for j in range(dim)])
            for i in range(seq_len)
        ]

        # Attention projections: dim -> dim
        self.Wq = Linear(
            xavier_like_mat(dim, dim, denom=denom, seed=seed + 2),
            zero_bias(dim), name="Wq"
        )
        self.Wk = Linear(
            xavier_like_mat(dim, dim, denom=denom, seed=seed + 3),
            zero_bias(dim), name="Wk"
        )
        self.Wv = Linear(
            xavier_like_mat(dim, dim, denom=denom, seed=seed + 4),
            zero_bias(dim), name="Wv"
        )
        self.Wo = Linear(
            xavier_like_mat(dim, dim, denom=denom, seed=seed + 5),
            zero_bias(dim), name="Wo"
        )

        # FFN: dim -> ffn_dim -> dim
        self.ffn_l1 = Linear(
            xavier_like_mat(ffn_dim, dim, denom=denom, seed=seed + 6),
            zero_bias(ffn_dim), name="ffn_l1"
        )
        self.ffn_l2 = Linear(
            xavier_like_mat(dim, ffn_dim, denom=denom, seed=seed + 7),
            zero_bias(dim), name="ffn_l2"
        )
        self.relu = ReLU()

        # Output projection: dim -> vocab_size
        self.output_proj = Linear(
            xavier_like_mat(vocab_size, dim, denom=denom, seed=seed + 8),
            zero_bias(vocab_size), name="output"
        )

        # Cache for backward pass
        self._cache = {}

    # -- embedding ---------------------------------------------------------

    def embed(self, token_ids):
        """
        Embed tokens and add positional encoding.

        I: list of token ids (ints), length <= SEQ_LEN
        O: list of Vec (embedded + positional)
        """
        xs = []
        for i, tid in enumerate(token_ids):
            tok = self.token_emb[tid]
            pos = self.pos_emb[i]
            xs.append(tok + pos)
        return xs

    # -- attention block ---------------------------------------------------

    def attention_block(self, xs):
        """
        Single-head causal self-attention.

        I: list of Vec (hidden states)
        O: list of Vec (attended states)

        Every attention weight row sums to exactly 1.
        """
        Q = [self.Wq.forward(x) for x in xs]
        K = [self.Wk.forward(x) for x in xs]
        V = [self.Wv.forward(x) for x in xs]

        attn_out = self_attention(
            Q, K, V, causal=True, exp_depth=cfg.EXP_DEPTH
        )

        projected = [self.Wo.forward(a) for a in attn_out]
        return projected

    def attention_block_with_cache(self, xs):
        """
        Attention block that caches intermediates for backward pass.

        O: (output, cache_dict)
        """
        Q = [self.Wq.forward(x) for x in xs]
        K = [self.Wk.forward(x) for x in xs]
        V = [self.Wv.forward(x) for x in xs]

        # compute scores and weights manually to cache them
        scores = attention_scores(Q, K)
        mask = causal_mask(len(xs))
        masked_scores = apply_boolean_mask(scores, mask)
        weights = [softmax(row, cfg.EXP_DEPTH) for row in masked_scores]
        attn_out = attention_mix(weights, V)

        projected = [self.Wo.forward(a) for a in attn_out]

        cache = {
            "Q": Q, "K": K, "V": V,
            "scores": masked_scores,
            "weights": weights,
            "attn_out": attn_out,
            "input": xs,
        }

        return projected, cache

    # -- FFN block ---------------------------------------------------------

    def ffn_block(self, x):
        """
        Feed-forward network: linear -> relu -> linear.

        I: Vec (single position)
        O: Vec
        """
        h = self.ffn_l1.forward(x)
        h = self.relu.forward(h)
        h = self.ffn_l2.forward(h)
        return h

    # -- forward -----------------------------------------------------------

    def forward(self, token_ids):
        """
        Full forward pass.

        I: list of token ids
        O: list of logit Vec (one per position)
        """
        # embed
        xs = self.embed(token_ids)

        # attention + residual
        attn = self.attention_block(xs)
        xs = [attn[i] + xs[i] for i in range(len(xs))]

        # FFN + residual
        ffn_out = [self.ffn_block(x) for x in xs]
        xs = [ffn_out[i] + xs[i] for i in range(len(xs))]

        # output projection
        logits = [self.output_proj.forward(x) for x in xs]

        return logits

    def forward_with_cache(self, token_ids):
        """
        Forward pass caching all intermediates for backward.

        O: (logits, cache) where cache has all intermediate values
        """
        # embed
        xs_embed = self.embed(token_ids)

        # attention + residual (with cache)
        attn_out, attn_cache = self.attention_block_with_cache(xs_embed)
        xs_post_attn = [attn_out[i] + xs_embed[i] for i in range(len(xs_embed))]

        # FFN + residual (cache relu inputs)
        ffn_pre = [self.ffn_l1.forward(x) for x in xs_post_attn]
        ffn_relu = [self.relu.forward(h) for h in ffn_pre]
        ffn_out = [self.ffn_l2.forward(h) for h in ffn_relu]
        xs_post_ffn = [ffn_out[i] + xs_post_attn[i] for i in range(len(xs_post_attn))]

        # output projection
        logits = [self.output_proj.forward(x) for x in xs_post_ffn]

        self._cache = {
            "token_ids": token_ids,
            "xs_embed": xs_embed,
            "attn_cache": attn_cache,
            "attn_out": attn_out,
            "xs_post_attn": xs_post_attn,
            "ffn_pre": ffn_pre,
            "ffn_relu": ffn_relu,
            "ffn_out": ffn_out,
            "xs_post_ffn": xs_post_ffn,
            "logits": logits,
        }

        return logits, self._cache

    def forward_last_logits(self, token_ids):
        """
        Forward pass, return only last position logits.

        I: token ids
        O: Vec (logits for last position)
        """
        logits = self.forward(token_ids)
        return logits[-1]

    def forward_last_logits_with_cache(self, token_ids):
        """
        Forward with cache, return last logits.

        O: (last_logits, cache)
        """
        logits, cache = self.forward_with_cache(token_ids)
        return logits[-1], cache

    # -- parameters --------------------------------------------------------

    def parameters(self):
        """All trainable parameters as flat list."""
        params = []
        params.extend(self.Wq.parameters())
        params.extend(self.Wk.parameters())
        params.extend(self.Wv.parameters())
        params.extend(self.Wo.parameters())
        params.extend(self.ffn_l1.parameters())
        params.extend(self.ffn_l2.parameters())
        params.extend(self.output_proj.parameters())
        return params

    def zero_grad(self):
        """Zero all parameter gradients."""
        for p in self.parameters():
            p.zero_grad()

    # -- backward (delegated to separate functions) ------------------------

    def backward_from_output(self, grad_logits):
        """
        Backpropagate from logit gradients through entire model.

        I: list of Vec — gradient of loss w.r.t. each position's logits
        S: accumulates gradients into all parameters

        Requires forward_with_cache to have been called first.
        """
        from attention_backward import attention_backward

        cache = self._cache
        n = len(grad_logits)

        # backward through output projection
        grad_post_ffn = []
        for i in range(n):
            g = self.output_proj.backward(grad_logits[i])
            grad_post_ffn.append(g)

        # residual split: grad flows to both FFN output and xs_post_attn
        grad_ffn_out = list(grad_post_ffn)
        grad_xs_post_attn_from_res2 = list(grad_post_ffn)

        # backward through FFN (per position)
        grad_xs_post_attn_from_ffn = []
        for i in range(n):
            # backward through ffn_l2
            # need to re-set ffn_l2's cached input
            self.ffn_l2._last_input = cache["ffn_relu"][i]
            g = self.ffn_l2.backward(grad_ffn_out[i])

            # backward through relu
            self.relu._last_input = cache["ffn_pre"][i]
            g = self.relu.backward(g)

            # backward through ffn_l1
            self.ffn_l1._last_input = cache["xs_post_attn"][i]
            g = self.ffn_l1.backward(g)

            grad_xs_post_attn_from_ffn.append(g)

        # combine residual gradients
        grad_xs_post_attn = [
            grad_xs_post_attn_from_res2[i] + grad_xs_post_attn_from_ffn[i]
            for i in range(n)
        ]

        # residual split: grad flows to both attention output and xs_embed
        grad_attn_out = list(grad_xs_post_attn)
        grad_xs_embed_from_res1 = list(grad_xs_post_attn)

        # backward through attention
        attn_cache = cache["attn_cache"]
        grad_xs_embed_from_attn = attention_backward(
            grad_attn_out,
            attn_cache["Q"], attn_cache["K"], attn_cache["V"],
            attn_cache["weights"],
            self.Wq, self.Wk, self.Wv, self.Wo,
            attn_cache["input"],
        )

        # combine residual gradients
        grad_xs_embed = [
            grad_xs_embed_from_res1[i] + grad_xs_embed_from_attn[i]
            for i in range(n)
        ]

        # grad_xs_embed is gradient w.r.t. embedded input
        # embeddings are fixed in v1, so we stop here

    def backward_from_last(self, grad):
        """
        Backpropagate from last position logit gradient only.

        I: Vec — gradient w.r.t. last position logits
        S: accumulates into all parameter gradients

        Other positions get zero gradient.
        """
        cache = self._cache
        n = len(cache["logits"])

        grad_logits = [Vec.zero(len(grad))] * n
        grad_logits[n - 1] = grad

        self.backward_from_output(grad_logits)
```

## File 4: `examples/toy_llm/attention_backward.py`

```python
"""
Backward pass through self-attention.

Separated from model.py because attention backward is the most
complex component. Every gradient is exact VDR rational.
"""

from vdr.core import VDR
from vdr.linalg import Vec


def softmax_backward(grad_probs, probs):
    """
    Backward through softmax.

    grad_score[j] = probs[j] * (grad_probs[j] - dot(grad_probs, probs))

    I: grad_probs Vec (gradient w.r.t. softmax output),
       probs Vec (softmax output values)
    O: grad_scores Vec (gradient w.r.t. softmax input)

    All exact VDR arithmetic. No numerical approximation.
    """
    # dot(grad_probs, probs)
    dot_gp = VDR(0)
    for i in range(len(probs)):
        dot_gp = dot_gp + grad_probs[i] * probs[i]

    result = []
    for j in range(len(probs)):
        result.append(probs[j] * (grad_probs[j] - dot_gp))
    return Vec(result)


def attention_mix_backward(grad_out, weights, V):
    """
    Backward through attention weighted sum.

    Forward: output[i] = sum_j weights[i][j] * V[j]

    Backward:
        grad_weights[i][j] = dot(grad_out[i], V[j])
        grad_V[j] = sum_i weights[i][j] * grad_out[i]

    I: grad_out (list of Vec), weights (list of Vec), V (list of Vec)
    O: (grad_weights, grad_V)
       grad_weights: list of Vec (same shape as weights)
       grad_V: list of Vec (same shape as V)
    """
    n_q = len(grad_out)
    n_k = len(V)
    dim = len(V[0])

    # grad_weights[i][j] = grad_out[i] . V[j]
    grad_weights = []
    for i in range(n_q):
        row = []
        for j in range(n_k):
            row.append(grad_out[i].dot(V[j]))
        grad_weights.append(Vec(row))

    # grad_V[j] = sum_i weights[i][j] * grad_out[i]
    grad_V = [Vec.zero(dim) for _ in range(n_k)]
    for j in range(n_k):
        for i in range(n_q):
            w_ij = weights[i][j]
            contribution = grad_out[i].scale(w_ij)
            grad_V[j] = grad_V[j] + contribution

    return grad_weights, grad_V


def score_backward(grad_scores, Q, K):
    """
    Backward through score computation: S[i][j] = Q[i] . K[j]

    Backward:
        grad_Q[i] = sum_j grad_scores[i][j] * K[j]
        grad_K[j] = sum_i grad_scores[i][j] * Q[i]

    I: grad_scores (list of Vec), Q (list of Vec), K (list of Vec)
    O: (grad_Q, grad_K) each list of Vec
    """
    n_q = len(Q)
    n_k = len(K)
    dim = len(Q[0])

    grad_Q = [Vec.zero(dim) for _ in range(n_q)]
    grad_K = [Vec.zero(dim) for _ in range(n_k)]

    for i in range(n_q):
        for j in range(n_k):
            g = grad_scores[i][j]
            if g != VDR(0):
                grad_Q[i] = grad_Q[i] + K[j].scale(g)
                grad_K[j] = grad_K[j] + Q[i].scale(g)

    return grad_Q, grad_K


def attention_backward(grad_projected, Q, K, V, weights, Wq, Wk, Wv, Wo, xs_input):
    """
    Full backward through single-head causal self-attention.

    I: grad_projected — list of Vec, gradient w.r.t. Wo output
       Q, K, V — cached query/key/value vectors
       weights — cached attention weight rows (each sums to exactly 1)
       Wq, Wk, Wv, Wo — Linear layers
       xs_input — original input to attention block

    O: grad_input — list of Vec, gradient w.r.t. attention block input
    S: accumulates gradients into Wq, Wk, Wv, Wo parameters

    All gradients exact VDR rationals.
    """
    n = len(grad_projected)

    # Step 1: backward through Wo
    # Wo.forward(attn_out[i]) produced projected[i]
    # We need to set Wo's cached input for each position
    grad_attn_out = []
    for i in range(n):
        # the input to Wo was the attention mix output
        # we need to reconstruct it: attn_out[i] = sum_j weights[i][j] * V[j]
        attn_out_i = Vec.zero(len(V[0]))
        for j in range(len(V)):
            attn_out_i = attn_out_i + V[j].scale(weights[i][j])

        Wo._last_input = attn_out_i
        g = Wo.backward(grad_projected[i])
        grad_attn_out.append(g)

    # Step 2: backward through attention mix
    grad_weights, grad_V = attention_mix_backward(grad_attn_out, weights, V)

    # Step 3: backward through softmax (per row)
    grad_scores = []
    for i in range(n):
        gs = softmax_backward(grad_weights[i], weights[i])
        grad_scores.append(gs)

    # Step 4: causal mask — zero out gradients for masked positions
    # masked positions had scores set to -1000, softmax made them ~0
    # gradient through those positions is already ~0 from softmax_backward
    # no explicit mask needed in backward

    # Step 5: backward through score computation
    grad_Q, grad_K = score_backward(grad_scores, Q, K)

    # Step 6: backward through Wv projections
    grad_input_from_V = [Vec.zero(len(xs_input[0])) for _ in range(n)]
    for i in range(n):
        Wv._last_input = xs_input[i]
        g = Wv.backward(grad_V[i])
        grad_input_from_V[i] = g

    # Step 7: backward through Wq projections
    grad_input_from_Q = [Vec.zero(len(xs_input[0])) for _ in range(n)]
    for i in range(n):
        Wq._last_input = xs_input[i]
        g = Wq.backward(grad_Q[i])
        grad_input_from_Q[i] = g

    # Step 8: backward through Wk projections
    grad_input_from_K = [Vec.zero(len(xs_input[0])) for _ in range(n)]
    for i in range(n):
        Wk._last_input = xs_input[i]
        g = Wk.backward(grad_K[i])
        grad_input_from_K[i] = g

    # Step 9: combine all input gradients
    grad_input = []
    for i in range(n):
        g = grad_input_from_Q[i] + grad_input_from_K[i] + grad_input_from_V[i]
        grad_input.append(g)

    return grad_input
```
