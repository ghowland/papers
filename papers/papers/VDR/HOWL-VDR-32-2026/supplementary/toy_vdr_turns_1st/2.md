```python
"""
examples/toy_llm/train.py

Toy LLM training — exact VDR arithmetic throughout.

    from train import train

    model, losses = train(n_epochs=20)
    # losses is list of exact VDR rationals

Every loss value is an exact fraction. No float anywhere.
Every softmax sums to exactly 1. Every gradient is exact.
"""

from vdr.core import VDR
from vdr.linalg import Vec
from vdr.ml.softmax import softmax
from vdr.ml.optim import SGD

from data import build_dataset, one_hot_target
from model import ToyTransformer
import config as cfg


# -- loss computation -----------------------------------------------------

def mse_loss(pred, target):
    """
    Mean squared error between two Vec.

    MSE = (1/n) * sum((pred[i] - target[i])^2)

    I: pred Vec, target Vec (same length)
    O: VDR scalar — exact rational

        mse_loss(Vec([VDR(1,2), VDR(1)]), Vec([VDR(0), VDR(1)]))
        -> VDR(1, 8)
    """
    n = len(pred)
    total = VDR(0)
    for i in range(n):
        diff = pred[i] - target[i]
        total = total + diff * diff
    return total * VDR(1, n)


def mse_loss_grad(pred, target):
    """
    Gradient of MSE w.r.t. pred.

    d(MSE)/d(pred[i]) = (2/n) * (pred[i] - target[i])

    I: pred Vec, target Vec
    O: Vec — exact gradient

        mse_loss_grad(Vec([VDR(1,2), VDR(1)]), Vec([VDR(0), VDR(1)]))
        -> Vec([VDR(1, 2), VDR(0)])
    """
    n = len(pred)
    scale = VDR(2, n)
    result = []
    for i in range(n):
        result.append((pred[i] - target[i]) * scale)
    return Vec(result)


def softmax_then_mse(logits, target_vec, exp_depth=None):
    """
    Softmax + MSE loss in one call.

    I: logits Vec, target_vec Vec (one-hot), exp_depth (optional)
    O: (probs, loss, grad_logits)
       probs: Vec summing to exactly 1
       loss: VDR scalar
       grad_logits: Vec — gradient of loss w.r.t. logits (through softmax)
    """
    if exp_depth is None:
        exp_depth = cfg.EXP_DEPTH

    probs = softmax(logits, exp_depth)
    loss = mse_loss(probs, target_vec)

    # gradient of MSE w.r.t. probs
    grad_probs = mse_loss_grad(probs, target_vec)

    # gradient through softmax: use softmax Jacobian
    # grad_logits[j] = probs[j] * (grad_probs[j] - dot(grad_probs, probs))
    from attention_backward import softmax_backward
    grad_logits = softmax_backward(grad_probs, probs)

    return probs, loss, grad_logits


# -- training step --------------------------------------------------------

def train_step(model, context_ids, target_id, optimizer):
    """
    Single training step: forward -> loss -> backward -> update.

    I: model ToyTransformer
       context_ids: list of ints (length SEQ_LEN)
       target_id: int (target token)
       optimizer: SGD instance

    O: (loss, probs) — exact VDR rationals
    S: model parameters updated in place

        loss, probs = train_step(model, [0,1,2,3], 0, optimizer)
    """
    # zero gradients from previous step
    model.zero_grad()

    # forward with cache (needed for backward)
    last_logits, cache = model.forward_last_logits_with_cache(context_ids)

    # target as one-hot
    target_vec = one_hot_target(target_id, cfg.VOCAB_SIZE)

    # softmax + loss + gradient through softmax
    probs, loss, grad_logits = softmax_then_mse(last_logits, target_vec)

    # backward through model
    model.backward_from_last(grad_logits)

    # optimizer step
    optimizer.step()

    return loss, probs


def train_epoch(model, windows, optimizer):
    """
    Train one epoch over all windows.

    I: model, list of (context_ids, target_id), optimizer
    O: list of (loss, probs) per window — all exact

        results = train_epoch(model, windows, optimizer)
        epoch_loss = sum(r[0] for r in results) / len(results)
    """
    results = []
    for context_ids, target_id in windows:
        loss, probs = train_step(model, context_ids, target_id, optimizer)
        results.append((loss, probs))
    return results


def average_loss(results):
    """
    Compute average loss from epoch results.

    I: list of (loss, probs) tuples
    O: VDR — exact average

        avg = average_loss(results)
    """
    total = VDR(0)
    for loss, _ in results:
        total = total + loss
    return total * VDR(1, len(results))


# -- full training loop ---------------------------------------------------

def train(n_epochs=None, model=None, verbose=True):
    """
    Full training loop.

    I: n_epochs (default from config), model (default: fresh),
       verbose (print losses)
    O: (model, loss_history)
       loss_history: list of VDR — average loss per epoch, all exact

    Every loss is an exact fraction.
    Every softmax sums to exactly 1.
    Every gradient is exact.
    Every parameter update is exact.

        model, losses = train()
    """
    if n_epochs is None:
        n_epochs = cfg.N_EPOCHS

    # build dataset
    windows, vocab, inv_vocab, vocab_size = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)

    # override vocab size in case corpus differs from config
    assert vocab_size == cfg.VOCAB_SIZE, (
        f"Corpus has {vocab_size} tokens, config says {cfg.VOCAB_SIZE}"
    )

    # build model
    if model is None:
        model = ToyTransformer()

    # optimizer
    optimizer = SGD(model.parameters(), lr=cfg.LR)

    # training loop
    loss_history = []

    for epoch in range(1, n_epochs + 1):
        results = train_epoch(model, windows, optimizer)
        avg_loss = average_loss(results)
        loss_history.append(avg_loss)

        if verbose:
            # display loss as exact fraction
            frac = avg_loss.to_fraction()
            approx = float(avg_loss)
            print(f"Epoch {epoch:3d}: loss = {frac}  (~{approx:.6f})")

            # verify softmax sum-to-one every epoch
            for i, (loss_i, probs_i) in enumerate(results):
                prob_sum = VDR(0)
                for j in range(len(probs_i)):
                    prob_sum = prob_sum + probs_i[j]
                assert prob_sum == VDR(1), (
                    f"Epoch {epoch} window {i}: softmax sum = {prob_sum}, "
                    f"expected exactly 1"
                )

    if verbose:
        print(f"\nTraining complete. {n_epochs} epochs, "
              f"{len(windows)} windows/epoch.")
        print(f"Final loss: {loss_history[-1].to_fraction()}")

    return model, loss_history


# -- convenience ----------------------------------------------------------

def quick_train(n_epochs=5):
    """
    Quick training run for testing.

    O: (model, losses) — trained for fewer epochs

        model, losses = quick_train()
    """
    return train(n_epochs=n_epochs, verbose=True)


def train_silent(n_epochs=None, model=None):
    """
    Train without printing.

    O: (model, losses)

        model, losses = train_silent(20)
    """
    return train(n_epochs=n_epochs, model=model, verbose=False)
```

```python
"""
examples/toy_llm/generate.py

Toy LLM text generation — exact VDR arithmetic.

    from generate import generate_text

    generate_text(model, "the cat", max_tokens=6)
    # "the cat sat on the mat" (or similar, depending on training)

Every sampling decision uses exact rational CDF comparison.
Same seed = same output on any platform, any Python version, any OS.
"""

from vdr.core import VDR
from vdr.linalg import Vec
from vdr.ml.softmax import softmax
from vdr.ml.rng import VDRRandom

from data import build_vocab, invert_vocab, tokenize, detokenize
import config as cfg


# -- sampling strategies --------------------------------------------------

def sample_categorical(probs, rng):
    """
    Sample from exact probability distribution via CDF comparison.

    Probabilities are exact VDR rationals summing to exactly 1.
    Random threshold is exact rational from deterministic PRNG.
    Comparison is exact — no float rounding at decision boundary.

    I: probs Vec (sums to exactly 1), rng VDRRandom
    O: int — sampled index

        rng = VDRRandom(seed=1)
        idx = sample_categorical(Vec([VDR(1,4), VDR(3,4)]), rng)
    """
    threshold = rng.random()  # exact rational in [0, 1)
    cumulative = VDR(0)
    for i in range(len(probs)):
        cumulative = cumulative + probs[i]
        if cumulative > threshold:
            return i
    # numerical safety: return last index
    return len(probs) - 1


def sample_greedy(probs):
    """
    Greedy decoding: return index of highest probability.

    Ties broken by lowest index (first occurrence).
    Comparison is exact — no float ambiguity.

    I: probs Vec
    O: int — argmax index

        sample_greedy(Vec([VDR(1,4), VDR(3,4)])) -> 1
    """
    best_idx = 0
    best_val = probs[0]
    for i in range(1, len(probs)):
        if probs[i] > best_val:
            best_val = probs[i]
            best_idx = i
    return best_idx


def sample_top_k(probs, k, rng):
    """
    Top-k sampling: zero out all but k highest probabilities,
    renormalize, then sample.

    I: probs Vec, k int, rng VDRRandom
    O: int — sampled index

    Renormalized probabilities sum to exactly 1.

        rng = VDRRandom(seed=1)
        idx = sample_top_k(Vec([VDR(1,10), VDR(2,10), VDR(7,10)]), 2, rng)
    """
    n = len(probs)
    if k >= n:
        return sample_categorical(probs, rng)

    # find top-k indices by sorting (exact comparison)
    indexed = [(probs[i], i) for i in range(n)]
    indexed.sort(key=lambda pair: pair[0], reverse=True)
    top_indices = set(indexed[j][1] for j in range(k))

    # zero out non-top-k, collect top-k values
    filtered = []
    for i in range(n):
        if i in top_indices:
            filtered.append(probs[i])
        else:
            filtered.append(VDR(0))

    # renormalize: divide each by sum of kept values
    total = VDR(0)
    for i in range(n):
        total = total + filtered[i]

    if total == VDR(0):
        # degenerate: all top-k are zero — fall back to uniform over top-k
        for j in range(k):
            idx = indexed[j][1]
            filtered[idx] = VDR(1, k)
        return sample_categorical(Vec(filtered), rng)

    renormed = [filtered[i] / total for i in range(n)]
    return sample_categorical(Vec(renormed), rng)


def sample_nucleus(probs, p_threshold, rng):
    """
    Nucleus (top-p) sampling: keep smallest set of tokens whose
    cumulative probability exceeds p_threshold, renormalize, sample.

    I: probs Vec, p_threshold VDR (e.g. VDR(9,10) for p=0.9),
       rng VDRRandom
    O: int — sampled index

    Exact rational CDF comparison at every step.

        rng = VDRRandom(seed=1)
        idx = sample_nucleus(probs, VDR(9, 10), rng)
    """
    n = len(probs)

    # sort by probability descending (exact comparison)
    indexed = [(probs[i], i) for i in range(n)]
    indexed.sort(key=lambda pair: pair[0], reverse=True)

    # accumulate until exceeding threshold
    cumulative = VDR(0)
    keep_indices = set()
    for prob_val, idx in indexed:
        keep_indices.add(idx)
        cumulative = cumulative + prob_val
        if cumulative >= p_threshold:
            break

    # zero out non-kept
    filtered = []
    for i in range(n):
        if i in keep_indices:
            filtered.append(probs[i])
        else:
            filtered.append(VDR(0))

    # renormalize
    total = VDR(0)
    for i in range(n):
        total = total + filtered[i]

    if total == VDR(0):
        return sample_categorical(probs, rng)

    renormed = [filtered[i] / total for i in range(n)]
    return sample_categorical(Vec(renormed), rng)


# -- generation core ------------------------------------------------------

def generate_ids(model, prompt_ids, max_tokens=10, strategy="greedy",
                 seed=1, top_k=3, top_p=None):
    """
    Autoregressive token generation.

    I: model ToyTransformer
       prompt_ids: list of int (starting tokens)
       max_tokens: int (tokens to generate)
       strategy: "greedy" | "sample" | "top_k" | "nucleus"
       seed: int (for deterministic sampling)
       top_k: int (for top-k strategy)
       top_p: VDR (for nucleus strategy, e.g. VDR(9,10))

    O: list of int — prompt_ids + generated ids

    Every step:
      1. Forward pass on last SEQ_LEN tokens
      2. Softmax on last position -> exact probabilities summing to 1
      3. Sample or select next token
      4. Append and repeat

        ids = generate_ids(model, [0, 1], max_tokens=4)
    """
    rng = VDRRandom(seed=seed)
    ids = list(prompt_ids)

    for _ in range(max_tokens):
        # take last SEQ_LEN tokens as context
        context = ids[-cfg.SEQ_LEN:]

        # pad if shorter than SEQ_LEN (left-pad with first token)
        while len(context) < cfg.SEQ_LEN:
            context = [context[0]] + context

        # forward pass — last position logits
        logits = model.forward_last_logits(context)

        # softmax — exact, sums to 1
        probs = softmax(logits, cfg.EXP_DEPTH)

        # select next token
        if strategy == "greedy":
            next_id = sample_greedy(probs)
        elif strategy == "sample":
            next_id = sample_categorical(probs, rng)
        elif strategy == "top_k":
            next_id = sample_top_k(probs, top_k, rng)
        elif strategy == "nucleus":
            if top_p is None:
                top_p = VDR(9, 10)
            next_id = sample_nucleus(probs, top_p, rng)
        else:
            raise ValueError(f"Unknown strategy: {strategy}")

        ids.append(next_id)

    return ids


# -- text-level interface -------------------------------------------------

def generate_text(model, prompt="the cat", max_tokens=6, strategy="greedy",
                  seed=1, top_k=3, top_p=None, corpus=None):
    """
    Generate text from a prompt string.

    I: model ToyTransformer
       prompt: string (space-separated tokens from vocabulary)
       max_tokens: int
       strategy: "greedy" | "sample" | "top_k" | "nucleus"
       seed: int
       corpus: string (for vocabulary; default from config)

    O: string — full generated text including prompt

        text = generate_text(model, "the cat", max_tokens=6)
    """
    if corpus is None:
        corpus = cfg.CORPUS

    vocab = build_vocab(corpus)
    inv_vocab = invert_vocab(vocab)

    prompt_ids = tokenize_safe(prompt, vocab)
    all_ids = generate_ids(
        model, prompt_ids, max_tokens=max_tokens,
        strategy=strategy, seed=seed, top_k=top_k, top_p=top_p,
    )
    return detokenize(all_ids, inv_vocab)


def tokenize_safe(text, vocab):
    """
    Tokenize with error handling for unknown tokens.

    I: text string, vocab dict
    O: list of int

    Raises ValueError if token not in vocabulary.
    """
    ids = []
    for word in text.split():
        if word not in vocab:
            raise ValueError(
                f"Token '{word}' not in vocabulary. "
                f"Known tokens: {list(vocab.keys())}"
            )
        ids.append(vocab[word])
    return ids


# -- convenience functions ------------------------------------------------

def generate_greedy(model, prompt="the cat", max_tokens=6, corpus=None):
    """
    Greedy generation shorthand.

        text = generate_greedy(model, "the cat")
    """
    return generate_text(
        model, prompt, max_tokens=max_tokens,
        strategy="greedy", corpus=corpus,
    )


def generate_sampled(model, prompt="the cat", max_tokens=6, seed=1,
                     corpus=None):
    """
    Sampled generation shorthand.

        text = generate_sampled(model, "the cat", seed=42)
    """
    return generate_text(
        model, prompt, max_tokens=max_tokens,
        strategy="sample", seed=seed, corpus=corpus,
    )


def generate_top_k_text(model, prompt="the cat", max_tokens=6, k=3,
                        seed=1, corpus=None):
    """
    Top-k generation shorthand.

        text = generate_top_k_text(model, "the cat", k=2, seed=42)
    """
    return generate_text(
        model, prompt, max_tokens=max_tokens,
        strategy="top_k", top_k=k, seed=seed, corpus=corpus,
    )


def generate_nucleus_text(model, prompt="the cat", max_tokens=6,
                          p=None, seed=1, corpus=None):
    """
    Nucleus (top-p) generation shorthand.

        text = generate_nucleus_text(model, "the cat", p=VDR(9,10))
    """
    if p is None:
        p = VDR(9, 10)
    return generate_text(
        model, prompt, max_tokens=max_tokens,
        strategy="nucleus", top_p=p, seed=seed, corpus=corpus,
    )


# -- display helpers ------------------------------------------------------

def show_generation(model, prompt="the cat", max_tokens=6, strategy="greedy",
                    seed=1, top_k=3, top_p=None, corpus=None):
    """
    Generate and display with probability details.

    Prints each step: token, probability, and full distribution.

        show_generation(model, "the cat", max_tokens=4)
    """
    if corpus is None:
        corpus = cfg.CORPUS

    vocab = build_vocab(corpus)
    inv_vocab = invert_vocab(vocab)

    prompt_ids = tokenize_safe(prompt, vocab)
    rng = VDRRandom(seed=seed)
    ids = list(prompt_ids)

    print(f"Prompt: '{prompt}'")
    print(f"Strategy: {strategy}")
    print(f"Prompt ids: {prompt_ids}")
    print()

    for step in range(max_tokens):
        context = ids[-cfg.SEQ_LEN:]
        while len(context) < cfg.SEQ_LEN:
            context = [context[0]] + context

        logits = model.forward_last_logits(context)
        probs = softmax(logits, cfg.EXP_DEPTH)

        # display distribution
        print(f"Step {step + 1}:")
        print(f"  Context: {[inv_vocab[i] for i in context]}")
        print(f"  Distribution:")
        for i in range(len(probs)):
            token_str = inv_vocab.get(i, f"<{i}>")
            p_frac = probs[i].to_fraction()
            p_approx = float(probs[i])
            print(f"    {token_str:8s}: {p_frac}  (~{p_approx:.4f})")

        # select
        if strategy == "greedy":
            next_id = sample_greedy(probs)
        elif strategy == "sample":
            next_id = sample_categorical(probs, rng)
        elif strategy == "top_k":
            next_id = sample_top_k(probs, top_k, rng)
        elif strategy == "nucleus":
            if top_p is None:
                top_p = VDR(9, 10)
            next_id = sample_nucleus(probs, top_p, rng)
        else:
            next_id = sample_greedy(probs)

        print(f"  Selected: {inv_vocab[next_id]} (id={next_id})")
        print()

        ids.append(next_id)

    result = detokenize(ids, inv_vocab)
    print(f"Full output: '{result}'")
    return result
```
