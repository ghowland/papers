% ================================================================
% VDR-LLM-Prolog — Numbers and Mathematics Specification
% Comprehensive Builtin Specification for Numeric Types and
% Mathematical Operations
%
% Method: Comprehensive (OSO C17)
%   The whole pie: every numeric operation the system needs.
%   Sliced by: number type × operation domain × precision level
%
% Design principle: VDR exact fractions are primary. Integer and
% decimal types exist for performance and user expectation.
% All three coexist. The system selects the appropriate type
% based on the operation's requirements. The LLM can use any
% layer through the same command token interface.
%
% Layers (inner to outer):
%   1. VDR exact fractions — the ground truth
%   2. Arbitrary-precision integers — for counting, indexing, IDs
%   3. Decimal display — for human-facing output
%   4. Q-basis compressed — for transcendental constants
%   5. Functional remainder — for convergent series
%
% Every layer has IOSE declarations.
% Every operation between layers has a declared conversion
% boundary with exact error bounds.
% ================================================================


% ================================================================
% SECTION 1: NUMBER TYPE HIERARCHY
%
% The system has five numeric representations. They form a
% hierarchy from most exact to most compact. Every value can
% move between representations through declared conversions.
% ================================================================

% ── type declarations ──────────────────────────────────────────

number_type(vdr_fraction, primary,
    "Exact rational number as VDR triple [V, D, R].
     V: arbitrary-precision integer numerator.
     D: nonzero arbitrary-precision integer denominator.
     R: remainder (atomic integer, composite, or functional).
     When R=0, behaves exactly as rational p/q.
     When R≠0, carries exact unresolved structure.
     This is the ground truth. All other types convert to/from this").

number_type(integer, native,
    "Arbitrary-precision integer. Equivalent to VDR [V, 1, 0].
     Used for: counting, indexing, IDs, bit operations, modular
     arithmetic. No denominator overhead when denominator is 1.
     Every integer is a valid VDR fraction. Promotion is free").

number_type(decimal_display, derived,
    "String representation of a fraction at declared precision.
     NOT a numeric type for computation. A display format only.
     Generated by fraction_to_decimal(frac, digits).
     The generating fraction is always available. The decimal
     is a lossy view, not a value").

number_type(qbasis, compressed,
    "Single arbitrary-precision integer over shared power-of-two
     denominator: [p, 2^k, 0]. Used for transcendental constants
     (pi, e, ln2, sqrt2, zeta3, and 17 others in Q335 basis).
     Addition of two Q-basis values with same k is one integer
     addition. Multiplication produces result in Q(2k) basis
     requiring reprojection to Q(k) with bounded error").

number_type(functional_remainder, deferred,
    "Python callable f(depth) -> VDR. Produces exact rational
     at each depth via convergent series or Newton iteration.
     Not evaluated until explicitly resolved. Each depth is a
     complete exact value, not an approximation of a limit.
     Used for: sqrt(n), pi series, exp(x), log(x), sin(x)").


% ── type relationships ─────────────────────────────────────────

% Promotion: lossless conversion upward
type_promotes(integer, vdr_fraction, lossless,
    "integer N becomes [N, 1, 0]. Zero cost").

type_promotes(qbasis, vdr_fraction, lossless,
    "qbasis [p, 2^k, 0] is already a valid VDR fraction").

% Projection: lossy conversion downward
type_projects(vdr_fraction, integer, lossy_if_not_integer,
    "floor, ceil, or round. Exact if denominator is 1").

type_projects(vdr_fraction, decimal_display, lossy,
    "Truncation at declared digit count. Generating fraction
     retained. Declared conversion boundary").

type_projects(vdr_fraction, qbasis, bounded_loss,
    "Round to nearest p/2^k. Error bounded by 2^(-k-1).
     For k=335, error is 10^66 times smaller than Planck length").

% Resolution: functional remainder to concrete
type_resolves(functional_remainder, vdr_fraction, at_declared_depth,
    "resolve(fn, depth=N) produces exact rational VDR.
     Each depth doubles correct digits (Newton) or adds
     fixed digits (series). No limit concept inside VDR").


% ── type selection rules ───────────────────────────────────────

% When to use which type
rule(select_number_type(integer, Operation) :-
    operation_domain(Operation, Domain),
    member(Domain, [counting, indexing, modular_arithmetic,
                    bit_operations, combinatorics, id_generation]),
    note("Use integer when denominator is always 1")).

rule(select_number_type(vdr_fraction, Operation) :-
    operation_domain(Operation, Domain),
    member(Domain, [general_arithmetic, linear_algebra,
                    probability, statistics, exact_comparison,
                    provenance_weight, training_parameter]),
    note("Use VDR fraction for exact rational computation")).

rule(select_number_type(qbasis, Operation) :-
    operation_domain(Operation, Domain),
    member(Domain, [transcendental_constants, compressed_weights,
                    pi_computation, exp_computation]),
    note("Use Q-basis for transcendental constant manipulation")).

rule(select_number_type(functional_remainder, Operation) :-
    operation_domain(Operation, Domain),
    member(Domain, [sqrt_computation, series_evaluation,
                    newton_iteration, convergent_approximation]),
    note("Use functional remainder for deferred exact computation")).


% ================================================================
% SECTION 2: VDR FRACTION CORE
%
% The primary numeric type. Every other type converts to/from this.
% Based on VDR-1 exact arithmetic with VDR-2/VDR-3 extensions.
%
% The triple [V, D, R]:
%   V = integer numerator (settled part in D-frame)
%   D = nonzero integer denominator (the frame)
%   R = remainder (exact unresolved structure)
%
% When R = 0: closed object, behaves as rational V/D
% When R != 0: active object, carries exact structure
% ================================================================

% ── structure ──────────────────────────────────────────────────

vdr_structure(triple, [
    slot(v, integer, "Numerator. Arbitrary precision. Settled component"),
    slot(d, integer_nonzero, "Denominator. Defines the frame. Never zero"),
    slot(r, remainder, "Exact unresolved structure. May be zero")
]).

remainder_form(atomic, "Single integer r. Simplest active form").
remainder_form(composite, "Integer base r plus finite list of child VDR triples.
    r + X1 + X2 + ... + Xn. Each Xi is a valid VDR. n is finite").
remainder_form(functional, "Callable f(depth) -> VDR with name string.
    Not evaluated until resolve() called. Each depth exact").

object_state(closed, "R = 0. Scalar projection Pi = V/D. Rational number").
object_state(active, "R != 0. Carries exact structure beyond V/D frame").


% ── normalization ──────────────────────────────────────────────

% Deterministic canonical form. Idempotent. Required for equality.
normalization_step(1, sign_convention,
    "If D < 0, negate both V and D").
normalization_step(2, gcd_reduction,
    "For closed nodes: divide V and D by gcd(|V|, |D|)").
normalization_step(3, child_normalization,
    "Normalize every child before parent. Bottom-up").
normalization_step(4, atomic_consolidation,
    "Combine all integer contributions at same level into one base").
normalization_step(5, canonical_ordering,
    "Sort children by denominator magnitude, then V, then R").
normalization_step(6, same_d_merge,
    "Closed children sharing denominator are added. Zero-sum removed").
normalization_step(7, closed_preference,
    "If remainder normalizes to zero, settle to closed form").


% ── equality ───────────────────────────────────────────────────

equality_type(structural, "Slot-by-slot recursive match. [1,2,0] != [2,4,0]").
equality_type(normalized_value, "Match after normalization. [1,2,0] == [2,4,0].
    Used by Python == operator and by Prolog unification of fractions").

% ── closed arithmetic invariants ───────────────────────────────

invariant(closed_addition_identity,
    "x + [0, 1, 0] = x for all closed x").
invariant(closed_multiplication_identity,
    "x * [1, 1, 0] = x for all closed x").
invariant(closed_additive_inverse,
    "x + (-x) = [0, 1, 0] for all closed x").
invariant(closed_multiplicative_inverse,
    "x * reciprocal(x) = [1, 1, 0] for all closed x where V != 0").
invariant(closed_commutativity,
    "x + y = y + x and x * y = y * x for all closed x, y").
invariant(closed_associativity,
    "(x + y) + z = x + (y + z) and (x * y) * z = x * (y * z)").
invariant(closed_distributivity,
    "x * (y + z) = x*y + x*z for all closed x, y, z").

% ================================================================
% SECTION 5: LIFT AND REBASE
%
% Lift: transport remainder when denominator frame changes.
% Rebase: change top-level denominator preserving exact value.
% ================================================================

builtin(vdr_lift, vdr_structure_ops,
    iose(
        [remainder, integer],
        [remainder],
        []),
    [pure, deterministic, bounded],
    "Remainder transport operator.
     lift(r, k) = k*r for atomic remainder.
     lift(r + X1..Xn, k) = k*r + lift(X1,k) + ... + lift(Xn,k).
     lift([V,D,R], k) = [k*V, D, lift(R,k)] — D preserved.
     Properties: identity at k=1, negation at k=-1,
     multiplicative composition, distributes over addition").

builtin(vdr_rebase, vdr_structure_ops,
    iose(
        [vdr_fraction, integer],
        [vdr_fraction],
        []),
    [pure, deterministic, bounded],
    "Change denominator frame.
     Closed rebase: if V*B/D is integer, result [V*B/D, B, 0].
     Active rebase: N=V*B, Q=N div D, S=N mod D.
       Result = [Q, B, [S,D,0] + lift(R,B)].
     Mismatch witness [S, D, 0] captures exact part that
     target denominator could not absorb.
     Preserves value equality, not structural equality").

builtin(vdr_scalar_projection, vdr_structure_ops,
    iose(
        [vdr_fraction],
        [vdr_fraction],
        []),
    [pure, deterministic, bounded],
    "Recursive evaluation Pi([V,D,R]) = (V + Pi(R)) / D.
     Produces exact closed rational.
     For closed objects, Pi([V,D,0]) = V/D.
     For active objects, recursively evaluates remainder.
     Functional remainders must be resolved first").

% ================================================================
% SECTION 10: Q-BASIS TRANSCENDENTAL OPERATIONS
%
% From MATH-4: 22 fundamental constants as integers over 2^335.
% Addition is one integer addition. Multiplication requires
% reprojection. Precision floor: 10^66 times Planck length.
% ================================================================

% ── constants ──────────────────────────────────────────────────

% All stored as qbasis(name, numerator_integer, exponent)
% The actual numerator is a ~102-digit integer.

% ================================================================
% SECTION 18: TYPE DISPATCH AND AUTOMATIC PROMOTION
%
% The system automatically selects the appropriate computation
% path based on operand types. Integer operations use the fast
% integer path. Mixed integer-fraction operations promote the
% integer to VDR fraction. The LLM does not need to track types
% explicitly — the dispatch is transparent.
% ================================================================

% Dispatch rules for arithmetic
rule(arithmetic_dispatch(integer, integer, int_path) :-
    note("Both integers: use fast integer arithmetic.
          No denominator computation needed")).

rule(arithmetic_dispatch(integer, vdr_fraction, vdr_path) :-
    note("Mixed: promote integer to [N,1,0], use VDR arithmetic")).

rule(arithmetic_dispatch(vdr_fraction, integer, vdr_path) :-
    note("Mixed: promote integer to [N,1,0], use VDR arithmetic")).

rule(arithmetic_dispatch(vdr_fraction, vdr_fraction, vdr_path) :-
    note("Both fractions: use VDR arithmetic")).

rule(arithmetic_dispatch(qbasis, qbasis, qbasis_path) :-
    note("Both Q-basis same exponent: use integer addition")).

rule(arithmetic_dispatch(qbasis, vdr_fraction, vdr_path) :-
    note("Mixed Q-basis and fraction: promote qbasis to fraction")).

% Automatic promotion is lossless and invisible
rule(promote(integer, vdr_fraction, Value, Promoted) :-
    Promoted = vdr(Value, 1, 0),
    note("Lossless. No information lost. No boundary logged
          because integer IS a VDR fraction")).

rule(promote(qbasis, vdr_fraction, qbasis(P, K), Promoted) :-
    Promoted = vdr(P, K, 0),
    note("Lossless. qbasis IS a VDR fraction")).
