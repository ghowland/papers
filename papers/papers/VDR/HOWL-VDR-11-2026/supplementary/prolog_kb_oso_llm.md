% SECTION 1: FOUNDATIONAL AXIOMS
% From: K1, K2, C2, C1
% Operations is fundamentally about control.
% Without control no other efficiency is possible.
% ================================================================

axiom(operations_is_control,
    "Operations is fundamentally about control — the ability to
     gather information from and change an environment").

axiom(control_precedes_efficiency,
    "Without control no other efficiency is possible.
     Establish control before optimizing").

axiom(control_requires_information,
    "Control requires knowing the present state. You cannot
     change what you cannot observe").

axiom(control_requires_agency,
    "Control requires the ability to effect change. Observation
     without agency is monitoring, not operations").

% The two halves of control
fact(control_component(observation, "gather information from environment")).
fact(control_component(agency, "change the environment")).

rule(has_control(System) :-
    has_observation(System),
    has_agency(System)).

rule(lacks_control(System, Reason) :-
    not(has_observation(System)),
    Reason = no_observation).
rule(lacks_control(System, Reason) :-
    not(has_agency(System)),
    Reason = no_agency).

% Resolve knowability fraction for a source type
rule(source_confidence(SourceType, Confidence) :-
    source_knowability(SourceType, Level),
    knowability_level(Level, Confidence, _)).

% Hearsay chain — trust degrades with each link
% From: D11, C47, C48
rule(chain_confidence([], fraction(1, 1))).
rule(chain_confidence([Link | Rest], Confidence) :-
    source_confidence(Link, LinkConf),
    chain_confidence(Rest, RestConf),
    Confidence is LinkConf * RestConf).

% Personal verification upgrades trust
% From: R12, C47
rule(verification_upgrade(Fact, OldConf, NewConf, Method) :-
    fact_confidence(Fact, OldConf),
    can_verify(Fact, Method),
    execute_verification(Method, confirmed),
    source_confidence(Method, MethodConf),
    NewConf is max(OldConf, MethodConf)).

% The hearsay rule: always prefer personal verification
% From: R12
rule(prefer_verified_over_hearsay(FactA, FactB, FactA) :-
    fact_verification_status(FactA, verified),
    fact_verification_status(FactB, unverified)).
rule(prefer_verified_over_hearsay(FactA, FactB, FactB) :-
    fact_verification_status(FactA, unverified),
    fact_verification_status(FactB, verified)).

% Should the system verify this fact before using it?
rule(should_verify(Fact) :-
    fact_source_type(Fact, SourceType),
    source_confidence(SourceType, Conf),
    Conf < fraction(90, 100)).

% Should the system refuse to assert this as a fact?
rule(should_not_assert_as_fact(Fact) :-
    fact_source_type(Fact, SourceType),
    source_confidence(SourceType, Conf),
    Conf < fraction(40, 100)).

% Data vs Logic knowability distinction
% From: D2, K3, K4
axiom(data_fully_knowable,
    "Stored data is fully knowable. You can read X and confirm
     X equals the stored value. Data is inspectable, durable,
     portable across contexts").

axiom(logic_partially_knowable,
    "Logic is only partially knowable. A program may halt or
     not, a decorator may change behavior invisibly, a stored
     procedure may have side effects not visible from the schema").

axiom(real_things_never_fully_knowable,
    "Physical things are never fully knowable. Sensors approximate.
     Temperature readings are samples, not the full thermal state.
     No sensor array is exhaustive").

% The decision rule: higher priority wins unless lower has 10x advantage
rule(priority_winner(A, B, A) :-
    priority(A, PA),
    priority(B, PB),
    PA >= PB * 10).

    priority(A, PA),
    priority(B, PB),
    PB >= PA * 10).

    PB < PA * 10).

% Cascade rule: check sources in priority order
rule(evidence_cascade([], none)).
rule(evidence_cascade([Source | Rest], Result) :-
    can_acquire(Source),
    acquire_evidence(Source, Evidence),
    Evidence \= none,
    Result = evidence(Source, Evidence)).
rule(evidence_cascade([Source | Rest], Result) :-
    (not(can_acquire(Source)) ; acquire_evidence(Source, none)),
    evidence_cascade(Rest, Result)).

    inputs([raw_token_stream]),

    side_effects([none]),

    inputs([dotted_path]),
    outputs([integer_id]),
    side_effects([id_assigned_if_new]),
    properties([operational, stable_across_sessions]))).

% IOSE validation rules

% Output type of producer must match input type of consumer
rule(iose_type_compatible(ProducerNode, ConsumerNode) :-
    iose(ProducerNode, _, outputs(Outs), _, _),
    iose(ConsumerNode, inputs(Ins), _, _, _),
    output_types(Outs, OutTypes),
    input_types(Ins, InTypes),
    compatible_types(OutTypes, InTypes)).

% Composite IOSE: external SE is union of children's SE
% minus internal data flows
rule(composite_side_effects(CompositeNode, ExternalSE) :-
    iose(CompositeNode, _, _, _, properties(Props)),
    member(composite, Props),
    children_of(CompositeNode, Children),
    findall(SE, (member(C, Children), iose(C, _, _, side_effects(SE), _)), AllSE),
    flatten(AllSE, FlatSE),
    remove_internal_flows(FlatSE, Children, ExternalSE)).

% Black-box rule: any IOSE node can be treated as opaque
rule(black_box(Node, Interface) :-
    iose(Node, Inputs, Outputs, SideEffects, _),
    Interface = iose_interface(Inputs, Outputs, SideEffects)).


% ================================================================
% SECTION 5: DATA PRIMACY RULES
% From: R8, K4, K5, K6, K7, D2
% Data is king. Logic is a replaceable shell.
% Data survives goal changes. Logic dies with its goals.
% ================================================================

axiom(data_primary_over_logic,
    "Data is more trustworthy, durable, and portable than logic.
     When data and logic conflict, trust the data and fix the logic").

axiom(data_survives_goals,
    "Goals change frequently. Logic dies with goals.
     Data survives because it describes what IS, not what to DO").

axiom(tool_logic_outlives_goal_logic,
    "Logic driven by data (tool-logic) outlives logic driven by
     a specific purpose (goal-logic) because data provides the
     context that keeps the logic relevant").

% Decision rules from data primacy
rule(trust_data_over_logic(Data, Logic, trust_data) :-
    fact_source_type(Data, data_source),
    fact_source_type(Logic, logic_source),
    data_and_logic_conflict(Data, Logic)).

% Architecture rule: IOSE interfaces carry data, logic stays inside
rule(data_at_interfaces(Node) :-
    iose(Node, inputs(I), outputs(O), _, _),
    all_data_typed(I),
    all_data_typed(O)).

% No logic in data store
% From: C54, R14
rule(no_logic_in_data_store(DataStore) :-
    data_store(DataStore),
    not(contains_stored_procedures(DataStore)),
    not(contains_triggers(DataStore))).

% Violation detection
rule(logic_in_data_store_violation(DataStore, Violation) :-
    data_store(DataStore),
    (contains_stored_procedures(DataStore),
     Violation = stored_procedures ;
     contains_triggers(DataStore),
     Violation = triggers)).


% ================================================================
% SECTION 6: COMPREHENSIVE VS AGGREGATED
% From: D3, C17, C18, C16, C15, K8, K9
% Build top-down from the whole. Subdivide preserving total volume.
% Aggregated systems trend toward inconsistency.
% ================================================================

axiom(comprehensive_over_aggregated,
    "Build comprehensive systems top-down from the whole,
     not aggregated systems bottom-up from pieces.
     Comprehensive systems trend toward consistency.
     Aggregated systems trend toward inconsistency").

axiom(slicing_the_pie,
    "Every subdivision must preserve the total volume.
     No information lost. No gaps. No overlaps.
     The parts must sum to the whole").

% Check if a system is comprehensive
rule(is_comprehensive(System) :-
    has_whole_definition(System),
    subdivisions(System, Parts),
    parts_cover_whole(Parts, System),
    no_gaps(Parts),
    no_overlaps(Parts)).

% Check if a system is aggregated
rule(is_aggregated(System) :-
    not(has_whole_definition(System)),
    components(System, Parts),
    not(parts_cover_whole(Parts, System))).

% The KB tree should be comprehensive
rule(kb_tree_comprehensive :-
    root_kb(Root),
    all_kbs(AllKBs),
    forall(member(KB, AllKBs),
        (has_path(KB) ; KB = Root)),
    forall(member(KB, AllKBs),
        (parent_exists(KB) ; KB = Root))).

% Aggregation warning: detect orphan KBs
rule(orphan_kb(KB) :-
    kb_exists(KB),
    KB \= root,
    not(parent_exists(KB))).

% Aggregation warning: detect uncategorized data sources
rule(uncategorized_source(Source) :-
    data_source(Source),
    not(corpus_category_assigned(Source, _))).


% ================================================================
% SECTION 7: IDEMPOTENCY
% From: C36, R3, K14
% f(f(x)) = f(x). Same result regardless of starting state.
% Makes automation safe to re-run.
% ================================================================

axiom(idempotency_enables_automation,
    "Idempotent operations converge to the desired state regardless
     of the starting state. This makes automation safe to re-run
     after failure, interruption, or uncertainty about current state").

% Tag operations as idempotent or not
fact(idempotent(session_restore)).
fact(idempotent(session_reset)).
fact(idempotent(kb_assert_existing_fact)).
fact(idempotent(kb_retract_nonexistent_fact)).
fact(idempotent(counter_set)).
fact(idempotent(lock_release_on_free_lock)).
fact(idempotent(bitset_set_already_set)).
fact(idempotent(deployment_of_same_checkpoint)).
fact(idempotent(rollback_to_same_checkpoint)).

fact(not_idempotent(counter_inc)).
fact(not_idempotent(queue_push)).
fact(not_idempotent(stack_push)).
fact(not_idempotent(ring_write)).
fact(not_idempotent(lru_push_new_key)).
fact(not_idempotent(kb_assert_new_fact)).

% Idempotent assert: only assert if not already present
rule(idempotent_assert(KB, Fact) :-
    (fact_exists(KB, Fact) -> true ; kb_assert(KB, Fact))).

% Idempotent retract: only retract if present
rule(idempotent_retract(KB, Fact) :-
    (fact_exists(KB, Fact) -> kb_retract(KB, Fact) ; true)).

% Safe re-run rule: for clone restarts
rule(safe_to_rerun(Operation) :-
    idempotent(Operation)).
rule(safe_to_rerun(Operation) :-
    not_idempotent(Operation),
    has_guard(Operation, Guard),
    guard_prevents_double_execution(Guard)).

% SECTION 10: TEMPORAL AWARENESS
% From: C41, K15
% All monitoring data is aged. You never truly know "now".
% Design for aged data. Track freshness explicitly.
% ================================================================

axiom(all_data_is_aged,
    "All monitoring data is aged. Sensor readings, metrics,
     query results — all represent the past, never the present.
     The delay may be milliseconds or minutes but is never zero").

% Freshness categories
fact(freshness_level(realtime, seconds(60),
    "Acquired within the last minute. Suitable for live triage")).
fact(freshness_level(recent, seconds(900),
    "Acquired within 15 minutes. Suitable for active investigation")).
fact(freshness_level(session_fresh, seconds(86400),
    "Acquired this session. Suitable for most analysis")).
fact(freshness_level(historical, seconds(999999999),
    "Any age with known timestamp. Suitable for trend analysis")).
fact(freshness_level(stale, seconds(0),
    "Age unknown or exceeds declared maximum. Use only as context")).

% Determine freshness of a piece of evidence
rule(evidence_freshness(Evidence, Level) :-
    evidence_acquired_at(Evidence, AcquiredAt),
    current_time(Now),
    Age is Now - AcquiredAt,
    freshness_level(Level, MaxAge, _),
    Age =< MaxAge).

% Should evidence be refreshed before use?
rule(needs_refresh(Evidence) :-
    evidence_acquired_at(Evidence, AcquiredAt),
    current_time(Now),
    Age is Now - AcquiredAt,
    max_staleness(Evidence, MaxAge),
    Age > MaxAge).

% Pipeline age accumulation
rule(total_pipeline_age(Evidence, TotalAge) :-
    evidence_acquired_at(Evidence, AcquiredAt),
    data_timestamp(Evidence, DataTimestamp),
    processing_time(Evidence, ProcessingTime),
    current_time(Now),
    DataAge is AcquiredAt - DataTimestamp,
    UseAge is Now - AcquiredAt,
    TotalAge is DataAge + ProcessingTime + UseAge).


% ================================================================
% SECTION 11: POPULATION STATISTICS
% From: R16, K16, E15
% Stats valid for populations, not individuals.
% You can predict how many disks fail, not which ones.
% ================================================================

axiom(population_stats_only,
    "Statistics are valid for populations, not for predicting
     individual events. You can predict 3 of 1000 disks will fail
     this month. You cannot predict which 3").

% Confidence scores are population predictions
rule(confidence_is_population_stat(Confidence) :-
    fraction(Confidence, N, D),
    note("Conclusions at confidence N/D are correct approximately
          N/D of the time across the population of conclusions
          at that confidence level. Any individual conclusion
          may be wrong")).

% Denominator growth is a population statistic
rule(denominator_growth_population(Step, MeanDenom, MaxDenom) :-
    findall(D, (parameter(P), param_denominator(P, Step, D)), Denoms),
    vdr_mean(Denoms, MeanDenom),
    vdr_max(Denoms, MaxDenom)).

% Clone drift thresholds are population-calibrated
rule(drift_threshold_is_population(Threshold, Metric) :-
    note("Threshold for metric is calibrated on observed population
          of clones. Individual clone may drift earlier or later")).


% ================================================================
% SECTION 12: FORCE MULTIPLIER SAFETY
% From: C66, K21, R11
% Automation amplifies both fixes and failures.
% Verify before automating. Minimum dependencies.
% ================================================================

axiom(force_multiplier_warning,
    "Automation is a force multiplier. It cuts problems efficiently
     AND cuts you efficiently. A bad automated process fails
     repeatedly, automatically, at scale").

% Verification before automation rules
rule(safe_to_automate(Process) :-
    manually_verified(Process),
    test_coverage(Process, Coverage),
    Coverage > fraction(90, 100),
    failure_mode_documented(Process)).

% Suspicious automation patterns
rule(automation_risk(repeated_clone_from_unverified_snapshot,
    "Clones from an unverified stable operator snapshot will
     reproduce the same errors automatically at scale")).
rule(automation_risk(inference_with_bad_rules,
    "Prolog rules that look consistent but encode wrong causal
     relationships produce systematically wrong conclusions
     with high confidence")).
rule(automation_risk(training_on_low_quality_data,
    "Training amplifies data quality issues. Low-quality training
     data produces low-quality outputs at the scale of all
     future interactions")).

% Overconfidence detection
rule(suspiciously_confident(Conclusion) :-
    conclusion_confidence(Conclusion, Conf),
    Conf > fraction(95, 100),
    evidence_count(Conclusion, Count),
    Count < 5).

% Minimum dependencies for operational components
% From: R11
rule(dependency_count_acceptable(Component) :-
    logic_classification(Component, operational),
    dependency_count(Component, N),
    N =< max_operational_dependencies).

fact(max_operational_dependencies, 5).

% SECTION 13: ONE WAY TO DO IT
% From: C33
% Production environment has single canonical method per task.
% ================================================================

axiom(one_way_to_do_it,
    "The production environment should have a single canonical
     method per task category. Not two ways to deploy, not three
     ways to query. One way. Known, documented, tested").

% Canonical method declarations
fact(canonical_method(assert_fact, kb_assert_command_token)).
fact(canonical_method(query_fact, kb_query_command_token)).
fact(canonical_method(execute_code, env_exec_through_environment_manager)).
fact(canonical_method(store_result, store_result_command_token)).
fact(canonical_method(read_file, fs_read_through_grant)).
fact(canonical_method(write_file, fs_write_through_grant)).
fact(canonical_method(fetch_external, net_fetch_through_grant)).
fact(canonical_method(snapshot_session, session_snapshot_command)).
fact(canonical_method(track_progress, counter_and_bitset)).

% Violation detection: more than one method for a task category
rule(one_way_violation(TaskCategory) :-
    findall(M, canonical_method(TaskCategory, M), Methods),
    length(Methods, N),
    N > 1).

% Shortcut detection: operation bypassing canonical method
rule(shortcut_detected(Operation, CanonicalMethod) :-
    operation_performed(Operation),
    operation_category(Operation, Category),
    canonical_method(Category, CanonicalMethod),
    operation_method(Operation, ActualMethod),
    ActualMethod \= CanonicalMethod).


% ================================================================
% SECTION 14: ALIGNMENT AND CONSISTENCY
% From: C22, C23, K23
% Internal consistency is prerequisite for alignment.
% Consistency is checkable. Alignment requires judgment.
% ================================================================

axiom(consistency_prereq_alignment,
    "Internal consistency is a prerequisite for alignment but
     not equivalent to it. A system can be internally consistent
     (all parts agree) but not aligned (the parts work at cross
     purposes to the goal). Both are required. Neither suffices").

% Internal consistency checks (automated, exact)
rule(internally_consistent(System) :-
    all_iose_contracts_satisfied(System),
    all_type_interfaces_match(System),
    all_constraints_pass(System),
    no_contradictions_in_kbs(System),
    all_provenance_chains_complete(System)).

% Alignment checks (require evaluation or human judgment)
rule(alignment_check(System, aligned) :-
    evaluation_passed(System, quality_benchmarks),
    evaluation_passed(System, safety_benchmarks),
    user_satisfaction_above(System, fraction(80, 100))).

rule(alignment_check(System, not_aligned) :-
    (not(evaluation_passed(System, quality_benchmarks)) ;
     not(evaluation_passed(System, safety_benchmarks)) ;
     not(user_satisfaction_above(System, fraction(80, 100))))).

% A system must pass both
rule(system_ready(System) :-
    internally_consistent(System),
    alignment_check(System, aligned)).


% ================================================================
% SECTION 15: AUTOMATION STRATEGY
% From: C3, C4, K19, K20, K28
% Remove classes of work. Less work tomorrow than today.
% Each class removed eliminates a scaling tax.

axiom(automation_removes_classes_of_work,
    "Automation is not about speeding up individual tasks.
     It is about removing entire classes of work so they never
     need to be done again. Each class removed eliminates
     a scaling tax").

axiom(less_work_tomorrow,
    "The discipline of operations is: less work tomorrow than
     today. Every action should reduce future work, not just
     complete current work").

axiom(manual_data_entry_breeds_errors,
    "Manual data entry breeds errors regardless of vigilance.
     If a human types it, a human will eventually mistype it.
     Automate data flow; do not manually transcribe").

% Classes of work that should be automated
fact(automate_class(data_quality_checking,
    "Corpus quality assessment should be automated pipeline,
     not manual inspection")).
fact(automate_class(constraint_verification,
    "All constraints should be checked automatically,
     not relied on humans to verify")).
fact(automate_class(provenance_recording,
    "Every derivation step should auto-assert its provenance,
     not require manual logging")).
fact(automate_class(drift_detection,
    "Clone drift should be detected by constraints,
     not by human observation")).
fact(automate_class(evidence_freshness_tracking,
    "Data staleness should be tracked automatically,
     not by human memory of when something was fetched")).
fact(automate_class(budget_enforcement,
    "Resource budgets should be enforced by counters and
     constraints, not by human discipline")).

% Rule: if you do it manually more than twice, automate it
rule(should_automate(Task) :-
    manual_execution_count(Task, Count),
    Count > 2,
    can_be_automated(Task)).


% ================================================================
% SECTION 16: PROMPT-TIME DECISION RULES
% Applied: how OSO principles govern the LLM's per-turn behavior
% ================================================================

% Evidence lookup order (from 90/9/0.9)
rule(prompt_time_lookup(Question, Answer, Source) :-
    lookup_order(Sources),
    evidence_cascade(Sources, evidence(Source, Answer))).

% Verify before asserting user claims
rule(prompt_time_user_fact(UserFact, Action) :-
    should_verify(UserFact),
    can_verify(UserFact, Method),
    Action = verify_then_assert(Method)).
rule(prompt_time_user_fact(UserFact, Action) :-
    should_verify(UserFact),
    not(can_verify(UserFact, _)),
    Action = assert_with_low_confidence(fraction(70, 100))).
rule(prompt_time_user_fact(UserFact, Action) :-
    not(should_verify(UserFact)),
    Action = assert_directly).

% Use KB (control model) over context window (understanding model)
rule(prompt_time_state_check(Query, Result) :-
    kb_can_answer(Query),
    kb_query(Query, Result),
    note("Prefer KB read over LLM memory of earlier turns")).

% Track conversation state in data primitives, not context
rule(prompt_time_state_tracking(StateItem, Method) :-
    is_countable(StateItem),
    Method = use_counter).
rule(prompt_time_state_tracking(StateItem, Method) :-
    is_recent_history(StateItem),
    Method = use_lru).
rule(prompt_time_state_tracking(StateItem, Method) :-
    is_pending_work(StateItem),
    Method = use_queue).
rule(prompt_time_state_tracking(StateItem, Method) :-
    is_completion_tracker(StateItem),
    Method = use_bitset).

% Do not assert LLM-generated content as high-confidence fact
rule(prompt_time_assertion_guard(Content, Guarded) :-
    content_source(Content, llm_generated),
    Guarded = assert_with_confidence(fraction(30, 100))).

% Explicit priority reorder (from R13)
rule(prompt_time_priority_change(OldPriority, NewPriority, Reason) :-
    note("When changing default priority ordering, state it
          explicitly so the decision is auditable"),
    log_priority_change(OldPriority, NewPriority, Reason)).

% SECTION 17: LIFECYCLE MANAGEMENT RULES
% Applied: how OSO principles govern training through retirement
% ================================================================

% Data quality is 90 priority in training
rule(lifecycle_data_quality_first :-
    lifecycle_priority(data_quality, P),
    P >= fraction(90, 100),
    note("Never sacrifice data quality for training speed")).

% Every data source gets knowability rating
rule(lifecycle_source_rated(Source) :-
    data_source(Source),
    source_knowability(Source, _)).

rule(lifecycle_source_unrated(Source) :-
    data_source(Source),
    not(source_knowability(Source, _))).

% Corpus must be comprehensive
rule(lifecycle_corpus_comprehensive(Corpus) :-
    corpus_categories(Corpus, Categories),
    forall(member(C, Categories),
        (document_count(Corpus, C, Count), Count > 0))).

% Training produces control model (must stay synced)
rule(lifecycle_checkpoint_is_control_model(Checkpoint) :-
    model_type(checkpoint_state, control, _),
    checkpoint_restorable(Checkpoint),
    note("Checkpoint must exactly reproduce the model state.
          This is a control model, not an approximation")).

% Feedback is hearsay — weight by agreement quality
rule(lifecycle_feedback_weighted(FeedbackRound, Weight) :-
    inter_annotator_agreement(FeedbackRound, Kappa),
    Weight is Kappa * fraction(70, 100),
    note("Annotator judgments are hearsay. Weight by agreement
          quality as a proxy for hearsay chain reliability")).

% Deployment requires both consistency and alignment
rule(lifecycle_deploy_ready(Checkpoint) :-
    internally_consistent(Checkpoint),
    alignment_check(Checkpoint, aligned),
    evaluation_passed(Checkpoint, safety_benchmarks),
    note("Must pass both internal consistency (automated) and
          alignment (evaluation-based) before deployment")).

% Monitoring is knowing the present (always aged)
rule(lifecycle_monitoring_aged :-
    note("All monitoring data is aged. Design dashboards and
          alerts for aged data. Never assume real-time")).

% Retirement preserves the control model for audit
rule(lifecycle_retirement_preserves_model(Model) :-
    archived(Model),
    queryable(Model),
    note("Retired model's KB is frozen but queryable.
          The control model persists for audit")).

% Canary is idempotent rollback
rule(lifecycle_canary_safe(Canary) :-
    rollback_target(Canary, Target),
    checkpoint_exists(Target),
    idempotent(deployment_of_same_checkpoint),
    note("Rollback to the same checkpoint is idempotent.
          The canary pattern relies on this")).

% SECTION 18: INFERENCE ORCHESTRATION RULES
% Applied: how OSO principles govern orchestrated inference

% Evidence cascade in inference (90/9/0.9 ordering)
rule(inference_evidence_order([
    kb_fact_read,
    pure_primitive_result,
    prolog_derivation,
    prometheus_live,
    deterministic_script,
    rest_api,
    web_search_result,
    llm_generated
])).

% Always acquire exact evidence before approximate
rule(inference_prefer_exact(ExactSource, ApproxSource) :-
    source_confidence(ExactSource, EC),
    source_confidence(ApproxSource, AC),
    EC > AC * fraction(10, 1),
    note("10x rule: prefer exact source when available")).

% Track hearsay chain length in inference provenance
rule(inference_chain_length(Evidence, Length) :-
    provenance_chain(Evidence, Chain),
    length(Chain, Length)).

rule(inference_long_chain_warning(Evidence) :-
    inference_chain_length(Evidence, Length),
    Length > 5,
    note("Hearsay chain longer than 5 links. Effective confidence
          degrades multiplicatively with each link")).

% Inference notebook is comprehensive (define goals first)
rule(inference_notebook_comprehensive(Notebook) :-
    has_goal(Notebook),
    has_evidence_dimensions(Notebook),
    has_budget(Notebook),
    note("Define what you're looking for before looking.
          Comprehensive, not aggregated investigation")).

% Inference uses control models (KB) not understanding models (LLM memory)
rule(inference_reads_control_model(Query, Result) :-
    kb_can_answer(Query),
    kb_query(Query, Result),
    note("Inference reads KB state (control model), not LLM
          context (understanding model). KB is authoritative")).

% Force multiplier check on inference conclusions
rule(inference_force_multiplier_check(Conclusion) :-
    (suspiciously_confident(Conclusion) ->
        note("Warning: conclusion confidence exceeds evidence depth.
              Force multiplier risk: wrong conclusion may be
              acted on with unwarranted confidence") ;
        true)).

% Idempotent evidence assertion in inference
rule(inference_assert_evidence(Notebook, Fact) :-
    idempotent_assert(Notebook, Fact),
    note("Evidence assertion is idempotent. Safe to re-assert
          after clone restart or notebook resumption")).

% Population stats apply to confidence
rule(inference_confidence_is_population :-
    note("A conclusion at 85/100 confidence means: across all
          conclusions scored at 85/100, approximately 85% are
          correct. This specific conclusion may be wrong")).
