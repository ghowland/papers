"""
examples/toy_llm/verify.py

Exact verification suite for the toy LLM.

    from verify import verify_all

    verify_all(model)
    # runs all 8 verification tests, prints results

Every test uses exact VDR comparison — no tolerances, no epsilons.
A test passes or fails by structural/value equality of exact rationals.
"""

from vdr.core import VDR
from vdr.linalg import Vec
from vdr.ml.softmax import softmax
from vdr.ml.optim import SGD
from vdr.ml.checkpoint import save_model, load_model_parameters

from data import build_dataset, one_hot_target
from model import ToyTransformer
from train import softmax_then_mse, train_step, train_silent
import config as cfg


# -- test 1: softmax sum-to-one -------------------------------------------

def verify_softmax_sum(model, windows=None):
    """
    Verify softmax sums to exactly 1 at every position, every window.

    Runs forward pass on every training window. For every position,
    computes softmax of logits and checks sum == VDR(1) exactly.

    I: model ToyTransformer, windows (optional, built from config)
    O: (passed: bool, details: str)

    Not approximately 1. Not 1 +/- epsilon. Exactly 1.
    """
    if windows is None:
        windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)

    violations = []

    for w_idx, (context_ids, target_id) in enumerate(windows):
        logits = model.forward(context_ids)
        for pos in range(len(logits)):
            probs = softmax(logits[pos], cfg.EXP_DEPTH)
            prob_sum = VDR(0)
            for i in range(len(probs)):
                prob_sum = prob_sum + probs[i]
            if prob_sum != VDR(1):
                violations.append(
                    f"window {w_idx} pos {pos}: sum = {prob_sum.to_fraction()}"
                )

    if violations:
        detail = "Softmax sum-to-one FAILED:\n  " + "\n  ".join(violations)
        return False, detail

    n_checks = len(windows) * cfg.SEQ_LEN
    return True, f"Softmax sum-to-one: {n_checks} checks, all exactly 1"


# -- test 2: gradient correctness ----------------------------------------

def verify_gradient_correctness(model=None):
    """
    Compare autodiff gradients against numerical gradients.

    For a single training step:
      1. Compute loss and backward (autodiff gradients)
      2. For a few parameters, compute numerical gradient via
         discrete derivative: (loss(w+h) - loss(w)) / h
      3. Compare — both exact rationals, must be exactly equal
         for sufficiently small h (they won't be equal in general
         because discrete derivative != continuous derivative,
         but for linear-in-parameter loss paths they match)

    For MSE loss with softmax, we verify the chain rule is consistent:
      numerical_grad ≈ autodiff_grad to high precision.

    I: model (optional, fresh if None)
    O: (passed: bool, details: str)
    """
    if model is None:
        model = ToyTransformer()

    windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)
    context_ids, target_id = windows[0]

    # step 1: autodiff gradient
    model.zero_grad()
    last_logits, cache = model.forward_last_logits_with_cache(context_ids)
    target_vec = one_hot_target(target_id, cfg.VOCAB_SIZE)
    probs, loss_base, grad_logits = softmax_then_mse(last_logits, target_vec)
    model.backward_from_last(grad_logits)

    # collect a few autodiff gradients from output projection weight
    params = model.output_proj.parameters()
    if not params:
        return False, "No parameters found in output projection"

    auto_grads = []
    for p in params[:3]:  # check first 3 parameters
        auto_grads.append(p.grad)

    # step 2: numerical gradient via finite difference
    h = VDR(1, 1000000)  # small step
    num_grads = []

    for p_idx in range(min(3, len(params))):
        p = params[p_idx]
        original_val = p.value

        # loss at w + h
        p.value = original_val + h
        logits_plus = model.forward_last_logits(context_ids)
        probs_plus = softmax(logits_plus, cfg.EXP_DEPTH)
        loss_plus = VDR(0)
        for i in range(len(probs_plus)):
            diff = probs_plus[i] - target_vec[i]
            loss_plus = loss_plus + diff * diff
        loss_plus = loss_plus * VDR(1, len(probs_plus))

        # numerical gradient
        num_grad = (loss_plus - loss_base) / h
        num_grads.append(num_grad)

        # restore
        p.value = original_val

    # step 3: compare — check that they are close
    # discrete derivative won't exactly match continuous for nonlinear loss,
    # but the difference should be small (proportional to h)
    details = []
    all_close = True
    for i in range(len(auto_grads)):
        ag = auto_grads[i]
        ng = num_grads[i]
        if ag is None:
            details.append(f"  param {i}: autodiff grad is None")
            all_close = False
            continue

        diff = ag - ng
        # for h = 1e-6, difference should be O(h) ≈ 1e-6
        # check |diff| < 1/100 (generous threshold for discrete vs continuous)
        diff_abs = diff if diff >= VDR(0) else VDR(0) - diff
        threshold = VDR(1, 100)

        if diff_abs > threshold:
            details.append(
                f"  param {i}: autodiff={ag.to_fraction()}, "
                f"numerical={ng.to_fraction()}, diff={diff.to_fraction()}"
            )
            all_close = False
        else:
            details.append(
                f"  param {i}: autodiff={ag.to_fraction()}, "
                f"numerical={ng.to_fraction()} — consistent"
            )

    if all_close:
        return True, "Gradient correctness: autodiff consistent with numerical\n" + "\n".join(details)
    else:
        return False, "Gradient correctness FAILED:\n" + "\n".join(details)


# -- test 3: weight update exactness -------------------------------------

def verify_weight_update(model=None):
    """
    Verify w_new = w_old - lr * grad exactly.

    After one training step:
      (w_old - w_new) / lr == grad

    All exact rational comparison.

    I: model (optional, fresh if None)
    O: (passed: bool, details: str)
    """
    if model is None:
        model = ToyTransformer()

    windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)
    context_ids, target_id = windows[0]

    optimizer = SGD(model.parameters(), lr=cfg.LR)

    # record w_old for a few parameters
    params = model.output_proj.parameters()
    w_old = [p.value for p in params[:5]]

    # train one step
    model.zero_grad()
    last_logits, cache = model.forward_last_logits_with_cache(context_ids)
    target_vec = one_hot_target(target_id, cfg.VOCAB_SIZE)
    probs, loss, grad_logits = softmax_then_mse(last_logits, target_vec)
    model.backward_from_last(grad_logits)

    # record gradients before step
    grads = [p.grad for p in params[:5]]

    # take optimizer step
    optimizer.step()

    # record w_new
    w_new = [p.value for p in params[:5]]

    # verify: (w_old - w_new) / lr == grad
    violations = []
    for i in range(len(w_old)):
        if grads[i] is None:
            violations.append(f"  param {i}: grad is None")
            continue

        computed_grad = (w_old[i] - w_new[i]) / cfg.LR
        if computed_grad != grads[i]:
            violations.append(
                f"  param {i}: (w_old-w_new)/lr = {computed_grad.to_fraction()}, "
                f"grad = {grads[i].to_fraction()}"
            )

    if violations:
        return False, "Weight update exactness FAILED:\n" + "\n".join(violations)

    return True, f"Weight update exactness: {len(w_old)} params verified, all exact"


# -- test 4: deterministic reproducibility --------------------------------

def verify_deterministic(n_epochs=3):
    """
    Same seed, same data, same result. Run twice, compare bit-identical.

    I: n_epochs to run
    O: (passed: bool, details: str)
    """
    # run 1
    model1, losses1 = train_silent(n_epochs=n_epochs)
    params1 = [p.value for p in model1.parameters()]

    # run 2 — fresh model from same seed
    model2, losses2 = train_silent(n_epochs=n_epochs)
    params2 = [p.value for p in model2.parameters()]

    # compare losses
    loss_match = True
    for i in range(len(losses1)):
        if losses1[i] != losses2[i]:
            loss_match = False
            break

    # compare parameters
    param_match = True
    for i in range(len(params1)):
        if params1[i] != params2[i]:
            param_match = False
            break

    if loss_match and param_match:
        return True, (
            f"Deterministic reproducibility: {n_epochs} epochs, "
            f"{len(params1)} params — bit-identical across runs"
        )
    else:
        details = []
        if not loss_match:
            details.append("Loss histories differ")
        if not param_match:
            details.append("Parameters differ")
        return False, "Deterministic reproducibility FAILED: " + "; ".join(details)


# -- test 5: loss monotonicity -------------------------------------------

def verify_loss_monotonicity(n_epochs=10, model=None):
    """
    With small enough learning rate, loss should generally decrease.

    Checks that final loss < initial loss (not strictly monotonic
    per step — SGD can oscillate with multiple windows).

    I: n_epochs, model (optional)
    O: (passed: bool, details: str)
    """
    model, losses = train_silent(n_epochs=n_epochs, model=model)

    if len(losses) < 2:
        return False, "Loss monotonicity: not enough epochs"

    first = losses[0]
    last = losses[-1]

    if last < first:
        return True, (
            f"Loss monotonicity: loss decreased from "
            f"{first.to_fraction()} to {last.to_fraction()} "
            f"over {n_epochs} epochs"
        )
    else:
        return False, (
            f"Loss monotonicity FAILED: first={first.to_fraction()}, "
            f"last={last.to_fraction()} (did not decrease)"
        )


# -- test 6: attention weights sum to 1 ----------------------------------

def verify_attention_weights(model=None):
    """
    Every row of attention weights sums to exactly 1.

    Runs forward_with_cache to access cached attention weights.

    I: model (optional)
    O: (passed: bool, details: str)
    """
    if model is None:
        model = ToyTransformer()

    windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)

    violations = []
    total_rows = 0

    for w_idx, (context_ids, target_id) in enumerate(windows):
        _, cache = model.forward_with_cache(context_ids)
        weights = cache["attn_cache"]["weights"]

        for row_idx, row in enumerate(weights):
            total_rows += 1
            row_sum = VDR(0)
            for i in range(len(row)):
                row_sum = row_sum + row[i]

            if row_sum != VDR(1):
                violations.append(
                    f"window {w_idx} row {row_idx}: "
                    f"sum = {row_sum.to_fraction()}"
                )

    if violations:
        return False, (
            "Attention weight sum FAILED:\n  " + "\n  ".join(violations)
        )

    return True, (
        f"Attention weight rows sum to 1: "
        f"{total_rows} rows checked, all exactly 1"
    )


# -- test 7: forward-backward roundtrip ----------------------------------

def verify_forward_backward_roundtrip():
    """
    Train one step, save parameters, load parameters,
    train same step again. Results bit-identical.

    I: (none)
    O: (passed: bool, details: str)
    """
    windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)
    context_ids, target_id = windows[0]

    # first run
    model1 = ToyTransformer()
    optimizer1 = SGD(model1.parameters(), lr=cfg.LR)

    model1.zero_grad()
    logits1, _ = model1.forward_last_logits_with_cache(context_ids)
    target_vec = one_hot_target(target_id, cfg.VOCAB_SIZE)
    probs1, loss1, grad1 = softmax_then_mse(logits1, target_vec)
    model1.backward_from_last(grad1)
    grads_run1 = [p.grad for p in model1.parameters()]
    optimizer1.step()
    params_after_run1 = [p.value for p in model1.parameters()]

    # second run — fresh model (same seed = same init)
    model2 = ToyTransformer()
    optimizer2 = SGD(model2.parameters(), lr=cfg.LR)

    model2.zero_grad()
    logits2, _ = model2.forward_last_logits_with_cache(context_ids)
    probs2, loss2, grad2 = softmax_then_mse(logits2, target_vec)
    model2.backward_from_last(grad2)
    grads_run2 = [p.grad for p in model2.parameters()]
    optimizer2.step()
    params_after_run2 = [p.value for p in model2.parameters()]

    # compare
    issues = []

    if loss1 != loss2:
        issues.append(f"Losses differ: {loss1.to_fraction()} vs {loss2.to_fraction()}")

    for i in range(len(grads_run1)):
        g1 = grads_run1[i]
        g2 = grads_run2[i]
        if g1 is None and g2 is None:
            continue
        if g1 != g2:
            issues.append(f"Gradient {i} differs")
            break

    for i in range(len(params_after_run1)):
        if params_after_run1[i] != params_after_run2[i]:
            issues.append(f"Parameter {i} differs after update")
            break

    if issues:
        return False, "Forward-backward roundtrip FAILED:\n  " + "\n  ".join(issues)

    return True, (
        f"Forward-backward roundtrip: loss, {len(grads_run1)} gradients, "
        f"{len(params_after_run1)} params — all bit-identical"
    )


# -- test 8: checkpoint roundtrip ----------------------------------------

def verify_checkpoint_roundtrip(model=None):
    """
    save_model -> load_model_parameters -> all parameters identical.

    I: model (optional, uses trained model)
    O: (passed: bool, details: str)
    """
    if model is None:
        model, _ = train_silent(n_epochs=3)

    # save
    state = save_model(model)

    # record originals
    original_params = [p.value for p in model.parameters()]

    # perturb all parameters to prove load actually restores
    for p in model.parameters():
        p.value = VDR(999, 1000)

    # verify perturbation worked
    perturbed = model.parameters()[0].value
    if perturbed == original_params[0]:
        return False, "Checkpoint roundtrip: perturbation did not change params"

    # load
    load_model_parameters(model, state)

    # compare
    restored_params = [p.value for p in model.parameters()]

    mismatches = 0
    for i in range(len(original_params)):
        if original_params[i] != restored_params[i]:
            mismatches += 1

    if mismatches > 0:
        return False, (
            f"Checkpoint roundtrip FAILED: "
            f"{mismatches}/{len(original_params)} params differ"
        )

    return True, (
        f"Checkpoint roundtrip: {len(original_params)} params "
        f"saved, perturbed, restored — all bit-identical"
    )


# -- run all tests --------------------------------------------------------

def verify_all(model=None, verbose=True):
    """
    Run all 8 verification tests.

    I: model (optional — some tests create fresh models)
       verbose: print results
    O: (all_passed: bool, results: list of (name, passed, detail))

        passed, results = verify_all(model)
    """
    tests = [
        ("1. Softmax sum-to-one",       lambda: verify_softmax_sum(model)),
        ("2. Gradient correctness",      lambda: verify_gradient_correctness()),
        ("3. Weight update exactness",   lambda: verify_weight_update()),
        ("4. Deterministic reproducibility", lambda: verify_deterministic(3)),
        ("5. Loss monotonicity",         lambda: verify_loss_monotonicity(10)),
        ("6. Attention weights sum to 1", lambda: verify_attention_weights(model)),
        ("7. Forward-backward roundtrip", lambda: verify_forward_backward_roundtrip()),
        ("8. Checkpoint roundtrip",       lambda: verify_checkpoint_roundtrip(model)),
    ]

    results = []
    all_passed = True

    if verbose:
        print("=" * 60)
        print("VDR Toy LLM — Exact Verification Suite")
        print("=" * 60)
        print()

    for name, test_fn in tests:
        try:
            passed, detail = test_fn()
        except Exception as e:
            passed = False
            detail = f"EXCEPTION: {type(e).__name__}: {e}"

        results.append((name, passed, detail))

        if not passed:
            all_passed = False

        if verbose:
            status = "PASS" if passed else "FAIL"
            print(f"[{status}] {name}")
            # indent detail lines
            for line in detail.split("\n"):
                print(f"       {line}")
            print()

    if verbose:
        print("=" * 60)
        n_pass = sum(1 for _, p, _ in results if p)
        n_total = len(results)
        print(f"Results: {n_pass}/{n_total} passed")
        if all_passed:
            print("All tests passed. Every value is exact.")
        else:
            print("Some tests failed. See details above.")
        print("=" * 60)

    return all_passed, results

"""

# -- parameter inspection -------------------------------------------------

def print_parameters(model, max_per_layer=8):
    """
    Print all model parameters with exact rational values.

    I: model ToyTransformer, max_per_layer (limit per component)

    Displays parameter name, shape hint, and exact fraction values.
    """
    components = [
        ("Wq", model.Wq),
        ("Wk", model.Wk),
        ("Wv", model.Wv),
        ("Wo", model.Wo),
        ("FFN L1", model.ffn_l1),
        ("FFN L2", model.ffn_l2),
        ("Output", model.output_proj),
    ]

    total = 0
    print("Model Parameters")
    print("-" * 50)

    for name, layer in components:
        params = layer.parameters()
        n = len(params)
        total += n
        print(f"\n{name} ({n} params):")
        for i, p in enumerate(params):
            if i >= max_per_layer:
                print(f"  ... ({n - max_per_layer} more)")
                break
            frac = p.value.to_fraction()
            print(f"  [{i}] {frac}")

    print(f"\nTotal parameters: {total}")

def print_attention_map(model, context_ids=None):
    """
    Print the attention weight matrix for a given context.

    Each cell is an exact rational. Rows sum to exactly 1.
    Causal mask visible as exact zeros in upper triangle.

    I: model, context_ids (default: first training window)
    """
    if context_ids is None:
        windows, _, _, _ = build_dataset(cfg.CORPUS, cfg.SEQ_LEN)
        context_ids = windows[0][0]

    inv_vocab = invert_vocab(vocab)

    _, cache = model.forward_with_cache(context_ids)
    weights = cache["attn_cache"]["weights"]

    tokens = [inv_vocab.get(tid, f"<{tid}>") for tid in context_ids]

    print("Attention Weights")
    print("-" * 50)
    print(f"Context: {tokens}")
    print()

    # header
    header = "         " + "".join(f"{t:>10s}" for t in tokens)
    print(header)

    # rows
    for i, row in enumerate(weights):
        row_str = f"{tokens[i]:8s} "
        row_sum = VDR(0)
        for j in range(len(row)):
            val = float(row[j])
            row_str += f"{val:10.4f}"
            row_sum = row_sum + row[j]
        row_str += f"  (sum={row_sum.to_fraction()})"
        print(row_str)

    print()

    # exact values
    print("Exact values:")
    for i, row in enumerate(weights):
        for j in range(len(row)):
            if row[j] != VDR(0):
                print(f"  [{i},{j}] {tokens[i]}->{tokens[j]}: "
                      f"{row[j].to_fraction()}")

# -- loss trajectory ------------------------------------------------------

def print_loss_trajectory(losses):
    """
    Print loss trajectory as exact fractions with approximate decimals.

    I: list of VDR loss values (from train())
    """
    print("Loss Trajectory")
    print("-" * 50)

    for i, loss in enumerate(losses):
        frac = loss.to_fraction()
        approx = float(loss)
        direction = ""
        if i > 0:
            if loss < losses[i - 1]:
                direction = " ↓"
            elif loss > losses[i - 1]:
                direction = " ↑"
            else:
                direction = " ="
        print(f"  Epoch {i + 1:3d}: {str(frac):>30s}  (~{approx:.6f}){direction}")

    if len(losses) >= 2:
        reduction = losses[0] - losses[-1]
        if losses[0] != VDR(0):
            pct_approx = float(reduction) / float(losses[0]) * 100
            print(f"\n  Total reduction: {reduction.to_fraction()} (~{pct_approx:.1f}%)")


# -- denominator report --------------------------------------------------

def denominator_report(model):
    """
    Report on denominator sizes across model parameters.

    Shows whether denominators are growing (would indicate
    computational cost increasing) or staying bounded.

    I: model ToyTransformer
    """
    params = model.parameters()

    denoms = []
    for p in params:
        v = p.value
        if hasattr(v, 'd'):
            denoms.append(v.d)

    if not denoms:
        print("Denominator report: no parameters found")
        return

    max_d = max(denoms)
    min_d = min(denoms)

    # count by magnitude
    buckets = {}
    for d in denoms:
        d_abs = abs(d)
        mag = len(str(d_abs))
        buckets[mag] = buckets.get(mag, 0) + 1

    print("Denominator Report")
    print("-" * 50)
    print(f"  Parameters: {len(denoms)}")
    print(f"  Min denominator: {min_d}")
    print(f"  Max denominator: {max_d}")
    print(f"  Max denominator digits: {len(str(abs(max_d)))}")
    print()
    print("  Distribution by digit count:")
    for mag in sorted(buckets.keys()):
        print(f"    {mag:3d} digits: {buckets[mag]:4d} params")


# -- combined inspection --------------------------------------------------

def full_inspection(model, losses=None):
    """
    Run all inspection functions on a model.

    I: model ToyTransformer, losses (optional, from train())
    """
    print("=" * 60)
    print("VDR Toy LLM — Full Inspection")
    print("=" * 60)
    print()

    print_embeddings(model)
    print()

    print_parameters(model, max_per_layer=4)
    print()

    print_attention_map(model)
    print()

    print_logits_and_probs(model)
    print()

    print_gradient_magnitudes(model)
    print()

    if losses:
        print_loss_trajectory(losses)
        print()

    denominator_report(model)
    print()

    print("=" * 60)
    print("Inspection complete. Every value above is exact.")
    print("=" * 60)
